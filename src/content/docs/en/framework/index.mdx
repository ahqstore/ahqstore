### Install the CLI (NodeJS)
<Steps>
  1. Install [NodeJS](https://nodejs.org/)
  2. Install your package manager
  .  <Tabs syncKey="pkg"> <TabItem label="npm"> <Code lang="sh" code="npm i -g npm"/> </TabItem> <TabItem label="yarn classic"> <Code lang="sh" code="npm i -g yarn"/> </TabItem> <TabItem label="yarn berry"> <Code lang="sh" code="npm i -g yarn; yarn version set berry"/> </TabItem> <TabItem label="pnpm"> <Code lang="sh" code="npm i -g pnpm"/> </TabItem> </Tabs>
</Steps>
<Tabs syncKey="pkg">
  <TabItem label="npm">
    ```sh
    npm i -g @ahqstore/cli
    ```
  </TabItem>
  <TabItem label="yarn classic">
    ```sh
    yarn global add @ahqstore/cli
    ```
  </TabItem>
  <TabItem label="yarn berry">
    ```sh
    yarn global add @ahqstore/cli
    ```
  </TabItem>
  <TabItem label="pnpm">
    ```sh
    pnpm add -g @ahqstore/cli
    ```
  </TabItem>
</Tabs>
