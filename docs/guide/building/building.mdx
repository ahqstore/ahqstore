import { Steps, Tabs, Tab } from "rspress/theme";

:::caution Beware
We recommend this only when there's no prebuilt binaries available. It'll also try to run the official installer
:::

<Steps>
  ### Step 1
  
  Clone the [GitHub repo](https://github.com/ahqsoftwares/tauri-ahq-store)

  ### Select your OS
  <Tabs>
  <Tab label="Windows">
    <div className="tw-mt-5" />
    <Steps>
      ### Install Rust
      
      [Download here](https://www.rust-lang.org/learn/get-started)
      
      ### Install NodeJS

      [Download here](https://nodejs.org/en)
      
      ### Run these commands

      ```sh
      npm i -g pnpm
      rustup install nightly
      pnpm install
      pnpm tauri build
      cd src-service
      cargo build --release
      cd ..
      ```

      Follow step 3 of the windows tutorial
    </Steps>
  </Tab>
  
  <Tab label="Linux">
    Run these commands
    
    ```sh
    curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh
    rustup install nightly
    sudo apt update
    sudo apt install libwebkit2gtk-4.1-dev build-essential curl wget file libssl-dev libayatana-appindicator3-dev librsvg2-dev
    curl -fsSL https://get.pnpm.io/install.sh | sh -
    pnpm install
    pnpm tauri build
    cd src-service
    cargo build --release
    cd ..
    ```
    Follow step 3 of the linux tutorial
  </Tab>
</Tabs>
</Steps>
